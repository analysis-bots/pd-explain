 16    def aggregate(self, func=None, *args, engine=None, engine_kwargs=None, **kwargs):
 17        """
 18        Aggregate using one or more operations over the specified axis.
 19
 20        :param func: Function to use for aggregating the data.
 21                     If a function, must either work when passed a DataFrame or when passed to DataFrame.apply.
 22
 23                     Accepted combinations are:
 24                     * function
 25                     * string function name
 26                     * list of functions and/or function names, e.g. [np.sum, 'mean']
 27                     * dict of axis labels -> functions, function names or list of such.
 28                     Can also accept a Numba JIT function with engine='numba' specified.
 29                     Only passing a single function is supported with this engine.
 30                     If the 'numba' engine is chosen, the function must be a user defined function with values and index
 31                      as the first and second arguments respectively in the function signature. Each groupâ€™s index will
 32                      be passed to the user defined function and optionally available for use.
 33        :param args: Positional arguments to pass to func.
 34        :param engine: * 'cython' : Runs the function through C-extensions from cython.
 35                       * 'numba' : Runs the function through JIT compiled code from numba.
 36                       * None : Defaults to 'cython' or globally setting compute.use_numba
 37        :param engine_kwargs:
 38        :param kwargs:
 39        :return: Explain Groupby Dataframe
 40        """
 41        from pd_explain.explainable_data_frame import ExpDataFrame
 42        result = ExpDataFrame(super().aggregate(func, engine, engine_kwargs, kwargs))
 43
 44        if hasattr(self, 'original'):
 45            original_result = self.original.aggregate(func, engine, engine_kwargs, kwargs)
 46            original_result.operation = GroupBy(source_df=None,
 47                                                source_scheme={},
 48                                                group_attributes=self.group_attributes,
 49                                                agg_dict=func,
 50                                                result_df=result,
 51                                                source_name=self.source_name)
 52
 53            return original_result
 54
 55        return result
